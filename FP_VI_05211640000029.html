Penjelasan : Dataset Suicide Rates Overview 1985 to 2016 merupakan dataset yang dikompilasi dari empat kumpulan data yaitu :

1. United Nations Development Program. (2018). Human development index (HDI).
2. World Bank. (2018). World development indicators: GDP (current US) by country:1985 to 2016.
3. [Szamil]. (2017). Suicide in the Twenty-First Century [dataset].

Keempat data ini dihubungkan oleh waktu dan tempat, dan dibangun menjadi dataset ini untuk menemukan hubungan antara social-ekonomi dengan peningkatan tingkat bunuh diri pada negara-negara di dunia. def suicides_rate_year_sex(data, year):
    #Filter data sesuai tahun
    data_year = data[data['year'] == year]
    
    data_suicide_rate = data_year.groupby(['country'])['suicides_no','population'].sum().reset_index()
    data_suicide_rate['suicide_rate'] = data_suicide_rate['suicides_no']/data_suicide_rate['population']
    data_male = data_year[data_year['sex']=='male'].groupby(['country','sex'])['suicides_no'].sum().reset_index()
    data_male = data_male.drop(['sex'],axis = 1)
    data_male.columns = ['country','suicides_no_male']
    data_join = pd.merge(data_suicide_rate, data_male, on='country')
    
    data_join['suicides_no_female'] = data_join['suicides_no'] - data_join['suicides_no_male']
    data_join['suicide_rate_male'] = data_join['suicides_no_male']/data_join['population']
    data_join['suicide_rate_female'] = data_join['suicides_no_female']/data_join['population']
    
    return data_join

def top_country_suicide_rates(years, top):
    # Initialize the matplotlib figure
    f, ax = plt.subplots(figsize=(16, 8))

    # Load the example car crash dataset
    year = years
    data_year = suicides_rate_year_sex(data, year)
    suicide = data_year.sort_values("suicide_rate", ascending=False)[:top]
    suicide_rate_max = data_year.suicide_rate.max()
    # Plot the total crashes
    sns.set_color_codes("pastel")
    sns.barplot(x="suicide_rate", y="country", data=suicide, label="Total", color="b")

    # Plot the crashes where alcohol was involved
    sns.set_color_codes("muted")
    sns.barplot(x="suicide_rate_male", y="country", data=suicide,label="male", color="b")

    # Add a legend and informative axis label
    ax.legend(ncol=2, loc="lower right", frameon=True)
    ax.set(xlim=(0, suicide_rate_max+0.00005), xlabel = 'suicide rate')
    ax.set_title("Negara dengan Suicide Rate paling tinggi pada tahun " +str(year))
    sns.despine(left=True, bottom=True)
    
top_country_suicide_rates(2015, 20)

def scat_gdp_suicide(year):
    fig= plt.figure(figsize=(16,10))

    data_year = data_suicide_gdp[data_suicide_gdp['year']==year].reset_index()
    X = data_year['gdp_per_capita']
    Y = data_year['suicide_rate']
    label = data_year['country']

    for i,j in enumerate(X):
        # look for the color based on vocabulary, if not found in vocubulary, then black is returned.
        plt.scatter(X[i] , Y[i])
        plt.annotate(label[i], # this is the text
                     (X[i],Y[i]), # this is the point to label
                     textcoords="offset points", # how to position the text
                     xytext=(0,10), # distance from text to points (x,y)
                     ha='center')
    plt.xlabel("gdp_per capita")
    plt.ylabel("suicide_rate")
    plt.title("Hubungan antara nilai ekonomi(GDP) negara dengan Suicide rite pada tiap negara tahun " +str(year))
    plt.ylim(data_year.suicide_rate.min()-0.00005, data_year.suicide_rate.max()+0.00005)
    plt.show()

def country_cumulative(all_country):
    fig, ax = plt.subplots(figsize=(16, 8))
    all_max = 0
    for country in all_country:
        data_country = data_all_year[data_all_year['country']== country]
        data_country['cum_sum'] = data_country['suicides_no'].cumsum()
        data_country.index = data_country.year
        country_max = data_country.cum_sum.max()
        if country_max > all_max:
            all_max = country_max
        ax = data_country.cum_sum.plot.area(stacked=False, label = country)

    ax.legend()
    ax.set_title('Jumlah Kumulatif Suicide per tahun pada tiap negara')
    plt.ylim(0, all_max+10000)
    
country_cumulative(['United States', 'Russian Federation','Japan','Republic of Korea'])

How : Yang pertama dilakukan adalah mengelompokkan data berdasarkan negara dan tahunnya. Setelah itu dilakukan penghitunggan cumulative suicides_no dan kemudian membuat area plot untuk setiap negara yang akan dibuat area plotnya

Analisis

Pada visualisasi yang telah dilakukan, kita dapat melihat jumlah kumulatif suicide dari negara United States, Japan , Korea dan Russia dari tahun ke tahun. Dapat dilihat bahwa Russia sempat mengalami peningkattan yang cukup tinggi tetapi mulai mengalami penurunan pada tahun 2000 ke atas. Sementara pada negara lain mengalami peningkatan yang cukup landai. Kecuali untuk negara Korea pada tahun 2005 ke atas yang mengalami peningkatan lebih tinggi dari sebelumnya.
